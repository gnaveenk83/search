package com.example.validator;

import org.apache.avro.Schema;
import org.apache.avro.generic.GenericDatumReader;
import org.apache.avro.generic.GenericRecord;
import org.apache.avro.io.Decoder;
import org.apache.avro.io.DecoderFactory;

import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Path;

public class AvroJsonValidatorApp {

    public static void main(String[] args) throws Exception {
        // Args: 0 = schema path, 1 = json path
        String schemaPath = args.length > 0 ? args[0] : "src/main/resources/user.avsc";
        String jsonPath   = args.length > 1 ? args[1] : "src/main/resources/input.json";

        // Load Avro schema
        String schemaStr = Files.readString(Path.of(schemaPath));
        Schema schema = new Schema.Parser().parse(schemaStr);

        // Load JSON payload from file
        String jsonStr = Files.readString(Path.of(jsonPath));

        System.out.printf("Validating file: %s%n", jsonPath);
        validate(schema, jsonStr);
    }

    private static void validate(Schema schema, String json) {
        try {
            var reader = new GenericDatumReader<GenericRecord>(schema);
            Decoder decoder = DecoderFactory.get().jsonDecoder(schema, json);
            GenericRecord record = reader.read(null, decoder);
            System.out.println("✅ Valid: " + record);
        } catch (IOException e) {
            System.err.println("❌ Invalid JSON: " + e.getMessage());
        }
    }
}
